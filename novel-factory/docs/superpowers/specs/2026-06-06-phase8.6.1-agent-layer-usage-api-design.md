### 4.1 Method shape

```python
# ContentWriterTools (infra/agent_system/agents/content_writer/tools.py)
def generate_chapter(self, chapter_num, context, writer_id=None) -> Dict:
    """旧 method, 签名不变, 内调用 _impl(record_usage=False)."""
    return self._impl_generate_chapter(chapter_num, context, writer_id, record_usage=False)

def generate_chapter_with_usage(
    self, chapter_num, context, writer_id=None,
) -> tuple[Dict, Dict[str, int]]:
    """New method — same result dict + real usage."""
    return self._impl_generate_chapter(chapter_num, context, writer_id, record_usage=True)

def _impl_generate_chapter(self, chapter_num, context, writer_id, record_usage):
    # ... existing variant/style/prompt setup ...
    if record_usage:
        response, usage = self.chat_with_usage(prompt=..., system=..., ...)
    else:
        response = self.chat(prompt=..., system=..., ...)
    result = self.parse_response(response, format_type="chapter")
    # ... metadata 注入 ...
    if record_usage:
        return result, usage
    return result
```

### 4.2 Polisher sum pattern (2 LLM)

`polish_chapter` 内部调 `optimize_dialogue_llm` + `adjust_pacing_llm` (2 LLM). `_impl` 走 2 chat_with_usage paths 并 sum usage:

```python
# PolisherTools (infra/agent_system/agents/polisher/tools.py)
def polish_chapter_with_usage(self, chapter_num, content, style_guide=None, reader_id=None):
    return self._impl_polish_chapter(chapter_num, content, style_guide, reader_id, record_usage=True)

def _impl_polish_chapter(self, chapter_num, content, style_guide, reader_id, record_usage):
    effective_reader = reader_id or self._current_reader_id or "A"
    polished = content
    issues: list = []
    usage_total = {"input_tokens": 0, "output_tokens": 0}

    # 步骤 1: LLM 优化对话
    try:
        if record_usage:
            polished, u1 = self.optimize_dialogue_llm_with_usage(polished, reader_id=effective_reader)
            usage_total["input_tokens"] += u1["input_tokens"]
            usage_total["output_tokens"] += u1["output_tokens"]
        else:
            polished = self.optimize_dialogue_llm(polished, reader_id=effective_reader)
    except Exception as e:
        issues.append({...})

    # 步骤 2: LLM 调整节奏
    try:
        if record_usage:
            polished, u2 = self.adjust_pacing_llm_with_usage(polished, reader_id=effective_reader)
            usage_total["input_tokens"] += u2["input_tokens"]
            usage_total["output_tokens"] += u2["output_tokens"]
        else:
            polished = self.adjust_pacing_llm(polished, reader_id=effective_reader)
    except Exception as e:
        issues.append({...})

    # 步骤 3: 规则去 AI 痕迹 (无 LLM, usage 不变)
    polished = self.remove_ai_gloss(polished)

    result = {
        "content": polished,
        "reader": effective_reader,
        "issues": issues,
        "chapter_num": chapter_num,
    }
    if record_usage:
        return result, usage_total
    return result

def optimize_dialogue_llm_with_usage(self, content, reader_id=None):
    return self._impl_optimize_dialogue(content, reader_id, record_usage=True)

def _impl_optimize_dialogue(self, content, reader_id, record_usage):
```

```python
class _UsageRecordingProvider(AIProvider):
    """Stub that records generate_with_usage calls + returns canned response."""
    def __init__(self, response="response"):
        super().__init__(ProviderConfig(api_key="sk-test", model="m"))
        self._response = response
        self.usage_calls: list = []

    def get_provider_name(self) -> str:

        return self._response

    def embed(self, text): return [0.0]
```
