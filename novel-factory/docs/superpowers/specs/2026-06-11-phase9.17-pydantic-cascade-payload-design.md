```python
# === Phase 9.17: cascade update WS payload schema ===

class CascadeUpdatePayload(BaseModel):
    """Phase 9.17: WS cascade.update payload schema (跟 RippleRollbackRequest 1:1 风格).

    替换 Phase 9.16 dict literal {ripple_id, cascade_node_count, cascade_edge_count,
    depth_reached, bfs_algorithm_version} — IDE 显式 types + Pydantic v2 runtime
    ValidationError 提前 (negative count / typo bfs_algorithm_version 立刻 fail).
    """
    ripple_id: str = Field(..., min_length=1)
    cascade_node_count: int = Field(..., ge=0)
    cascade_edge_count: int = Field(..., ge=0)
    depth_reached: int = Field(..., ge=0)
    bfs_algorithm_version: Literal["v1", "v2_weighted"]
```

**`dashboard/cascade_notifier.py:37` 改 typed param**:
```python
# 旧:
def notify_cascade_update(payload: dict[str, Any]) -> None:

# 新 (Phase 9.17):
def notify_cascade_update(payload: "CascadeUpdatePayload") -> None:
    # Runtime duck typing: Pydantic BaseModel instance 既是 mapping-like
    # (model_dump 序列化), 也支持 attribute access (payload.ripple_id).
    # 显式 isinstance check 1:1 跟 Phase 9.14 record_audit.
    if _ws_manager is None:
        logger.debug("cascade_notifier: ws_manager not set, skip broadcast")
        return
    # 防御: caller 传 dict 时 runtime ValidationError 兜底 (backward compat)
    if not isinstance(payload, CascadeUpdatePayload):
        try:
            payload = CascadeUpdatePayload.model_validate(payload)
        except Exception as e:
            logger.warning("cascade_notifier: invalid payload, skip: %s", e)
            return
    envelope = {"type": "cascade.update", "payload": payload.model_dump()}
    ws = _ws_manager
    try:
        result = ws(envelope)
        if asyncio.iscoroutine(result):
            try:
                loop = asyncio.get_event_loop()
                if loop.is_running():
                    asyncio.ensure_future(result)
                else:
                    loop.run_until_complete(result)
            except RuntimeError:
                asyncio.ensure_future(result)
    except Exception as e:
        logger.warning("cascade_notifier: broadcast failed: %s", e)
```

### 3.2 T2: `MAX_HANDLERS=50` guard in `useWorkflowSocket.js`

**Pattern** (`useWorkflowSocket.js:32, 175-190`):
```javascript

// Phase 9.17: guard 防止 handler 累积 leak (反复 mount/unmount / 脚本 bug)
const MAX_HANDLERS = 50;

export function onCascadeUpdate(handler) {

    console.warn(
      `[useWorkflowSocket] MAX_HANDLERS=${MAX_HANDLERS} reached, refusing to register cascade handler`
    );
    return;

  registeredHandlers.cascade.add(handler);
  ...
}
```
