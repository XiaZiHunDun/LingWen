### 1. Component: `CascadeRunsPanel.vue` (NEW, ~250L)

**Props** (单一 prop, 数据 scoped to current ripple):
```javascript
defineProps({
  rippleId: { type: String, required: true },
});
```

**State** (ref-based, 内部生命周期):
- `runs: ref([])` — sorted by `id` DESC (most recent first)
- `loading: ref(false)`, `error: ref(null)`
- `selectedRunId: ref(null)`, `replayedData: ref(null)`, `replayedFromId: ref(null)`, `replayedFromStartedAt: ref(null)`
- `replaying: ref(null)`, `cancelling: ref(null)` — track in-flight action per run id

**Template 骨架**:

### 3. API methods (api/index.js +30L)

```javascript
// === Phase 9.22: cascade_runs list + cancel API methods ===

/**
 * Phase 9.22: fetch historical cascade_runs for a ripple (Phase 9.20 endpoint).
 * Returns CascadeRun[] JSON array: [{ id, ripple_id, started_at, completed_at,
 *   max_depth, status, algorithm_version, ... }, ...] sorted by id DESC.
 */
export async function fetchCascadeRuns(rippleId, options = {}) {
  const params = new URLSearchParams();
  if (options.limit) params.set('limit', String(options.limit));
  if (options.offset) params.set('offset', String(options.offset));
  if (options.status) params.set('status', options.status);
  const qs = params.toString();
  return request(`/api/ripples/cascade/${encodeURIComponent(rippleId)}/runs${qs ? '?' + qs : ''}`);
}

/**
 * Phase 9.22: cancel a cascade run (Phase 9.21 endpoint).
 * Returns updated CascadeRun JSON. body: { reason?: string }.
 */
export async function cancelCascadeRun(rippleId, runId, reason = '') {
  return request(`/api/ripples/cascade/${encodeURIComponent(rippleId)}/runs/${runId}/cancel`, {
    method: 'POST',
    body: JSON.stringify({ reason }),
  });
}

/**
 * Phase 9.22: re-run BFS with a given max_depth for the Replay button.
 * Calls Phase 9.20 GET /api/ripples/cascade/{id}?max_depth=N&persist=false.
 * persist=false keeps cascade_runs table clean (replay is read-only, not a new run).
 * Returns cascade JSON (same shape as fetchRippleCascade response).
 */
export async function fetchCascadeWithDepth(rippleId, maxDepth) {
  return request(
    `/api/ripples/cascade/${encodeURIComponent(rippleId)}?max_depth=${encodeURIComponent(maxDepth)}&persist=false`
  );
}
```

Note: `fetchCascadeRuns` calls the existing Phase 9.20 `/api/ripples/cascade/{id}/runs` endpoint (new namespace, additive alongside legacy `/api/cvg/ripples/.../cascade`). The endpoint signature already supports `?status=` filter (Phase 9.20) — we pass it through but UI defaults to no filter (YAGNI, Phase 9.23 adds filter UI). `fetchCascadeWithDepth` is a new thin wrapper used by Replay; we don't extend `useRippleStore.fetchRippleCascade` signature (YAGNI, panel talks to api/index.js directly, store stays 0 改).

### 4. Replay flow

```javascript
async function replay(run) {
  if (replaying.value) return;
  replaying.value = run.id;
  try {
    // Call new api method `fetchCascadeWithDepth(rippleId, maxDepth)` (see §3)
    // which hits Phase 9.20 endpoint /api/ripples/cascade/{id}?max_depth=N&persist=false
    const data = await fetchCascadeWithDepth(props.rippleId, run.max_depth);
    replayedData.value = data;
    replayedFromId.value = run.id;
    replayedFromStartedAt.value = run.started_at;
    selectedRunId.value = run.id;
  } catch (e) {
    error.value = `Replay failed: ${e.message}`;
  } finally {
    replaying.value = null;
  }
}
```

**Decision (concrete)**: Replay uses the new thin api method `fetchCascadeWithDepth(rippleId, maxDepth)` (defined in §3) which calls Phase 9.20 `GET /api/ripples/cascade/{id}?max_depth=N&persist=false`. We do NOT extend `useRippleStore.fetchRippleCascade` signature (YAGNI — store stays 0 改, panel talks to api/index.js directly). We do NOT reuse the legacy Phase 9.15 `/api/cvg/ripples/{id}/cascade` endpoint because Phase 9.20's new endpoint has the `max_depth` query param and explicit `persist` flag (which we set to `false` to avoid polluting `cascade_runs` with replay reads).

### 5. Cancel flow

```javascript
async function cancel(run) {
  if (cancelling.value) return;
  cancelling.value = run.id;
  const prevStatus = run.status;
  // Optimistic update
  run.status = 'cancelled';
  try {
    await cancelCascadeRun(props.rippleId, run.id, '');
    // WS cascade.cancel event will arrive shortly and re-confirm
  } catch (e) {
    run.status = prevStatus;  // revert
    error.value = `Cancel failed: ${e.message}`;
  } finally {
    cancelling.value = null;
  }
}
```

### 6. WS handler integration

```javascript
import { onMounted, onBeforeUnmount } from 'vue';
import { registerCascadeHandler, unregisterCascadeHandler } from '../composables/useWorkflowSocket';

let cascadeHandler = null;

onMounted(async () => {
  await loadRuns();
  // Phase 9.22: register cascade handler for live updates
  cascadeHandler = (payload) => {
    if (payload.ripple_id !== props.rippleId) return;
    // Phase 9.17: cascade.update event has no run_id (only counts) → append new running row
    // Phase 9.21: cascade.cancel event has run_id + status='cancelled' → update existing row
    if (payload.status === 'cancelled' && payload.run_id != null) {
      const existing = runs.value.find((r) => r.id === payload.run_id);
      if (existing) existing.status = 'cancelled';
    } else {
      // New cascade.update → optimistic append new running row
      // (full data not in payload; will be re-fetched on next loadRuns)
      // For simplicity: re-fetch entire list on cascade.update
      loadRuns();
    }
  };
  registerCascadeHandler(cascadeHandler);
});

onBeforeUnmount(() => {
  if (cascadeHandler) unregisterCascadeHandler(cascadeHandler);
});
```

**Decision on cascade.update handling**: payload doesn't include run_id (Phase 9.17 schema has ripple_id, counts, depth, algorithm_version — no run_id). Simplest: re-fetch the list on cascade.update. This adds 1 HTTP call per cascade update but keeps state consistent. Phase 9.23+ can add run_id to payload if needed.

### 7. Status transition matrix (permissive YAGNI)

| 触发 | source | target | 行为 |
|------|--------|--------|------|
| Initial load | `fetchCascadeRuns` | runs[] | 填充 + sort by id DESC |
| WS cascade.update | payload.ripple_id match | runs[] | re-fetch entire list |
| WS cascade.cancel | payload.run_id match | run.status | 找到 row → status = "cancelled" |
| User click Replay | button click | replayedData | `fetchCascadeWithDepth(rippleId, run.max_depth)` → render |
| User click Cancel | button click | run.status | optimistic → "cancelled", 失败 revert |
