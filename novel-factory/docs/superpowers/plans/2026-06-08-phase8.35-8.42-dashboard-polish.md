```javascript
// eslint-rules/testid-class-sync.js — Phase 8.36
// 规则: .vue 元素加 data-testid="x" 必加 class 含 "x" (kebab-case 镜像)
//       加 class="...x..." 必加 data-testid="x"
// 形态: 镜像 Phase 8.33 no-class-selector-in-test.js pattern (本地 ESLint rule).
// 例外: VElement 0 渲染 (e.g. <script> / <template> root) 0 强;
//       顶层 <template> 0 强 (root 必是 1 element 但 element 自身 0 必加 testid).

'use strict'

module.exports = {
  meta: {
    type: 'problem',
    docs: {
      description:
        '强制 .vue 元素加 data-testid="x" 必加 class 含 "x", 反之亦然 (kebab-case 镜像)',
    },
    schema: [],
  },
  create(context) {
    return context.parserServices.defineTemplateBodyVisitor({
      // VAttribute visitor: 元素加 data-testid, 必加同 class
      "VAttribute[directive=false][key.name='data-testid']"(node) {
        const testid = node.value && node.value.value
        if (!testid) return
        const parent = node.parent
        const classAttr = (parent.attributes || []).find(
          (a) => a.key && a.key.name === 'class',
        )
        if (!classAttr || !classAttr.value || !classAttr.value.value) {
          context.report({
            node,
            message: `data-testid="${testid}" 必须加 class 镜像`,
          })
          return
        }
        if (!classAttr.value.value.split(/\s+/).includes(testid)) {
          context.report({
            node,
            message: `class 必须包含 "${testid}" (kebab-case 镜像 testid)`,
          })
        }
      },
      // VAttribute visitor: 元素加 class (无 testid), 必加同 testid (单 token kebab-case)
      "VAttribute[directive=false][key.name='class']"(node) {
        const cls = node.value && node.value.value
        if (!cls) return
        const parent = node.parent
        const hasTestid = (parent.attributes || []).some(
          (a) => a.key && a.key.name === 'data-testid',
        )
        if (hasTestid) return  // 上面 visitor 已 check
        // 找 class 中 kebab-case 单 token, 报告每个 (YAGNI: 不处理 multi-word)
        const tokens = cls.split(/\s+/).filter(Boolean)
        for (const t of tokens) {
          if (/^[a-z][a-z0-9-]*$/.test(t) && t.length > 2) {
            context.report({
              node,
              message: `class="${t}" 必加 data-testid="${t}"`,
            })
          }
        }
      },
    })
  },
}
```
