def _impl_polish_chapter(
    self, chapter_num, content, style_guide, reader_id, record_usage,
) -> Any:
    effective_reader = reader_id or self._current_reader_id or "A"
    polished = content
    issues: list = []
    usage_total = {"input_tokens": 0, "output_tokens": 0}

    # 步骤 1: LLM 优化对话
    try:
        if record_usage:
            polished, u1 = self._impl_optimize_dialogue(polished, effective_reader, record_usage=True)
            usage_total["input_tokens"] += u1["input_tokens"]
            usage_total["output_tokens"] += u1["output_tokens"]
        else:
            polished = self._impl_optimize_dialogue(polished, effective_reader, record_usage=False)
    except Exception as e:
        issues.append({"type": "dialogue_llm_fail", "severity": "P3", "issue": str(e)})

    # 步骤 2: LLM 调整节奏
    try:
        if record_usage:
            polished, u2 = self._impl_adjust_pacing(polished, effective_reader, record_usage=True)
            usage_total["input_tokens"] += u2["input_tokens"]
            usage_total["output_tokens"] += u2["output_tokens"]
        else:
            polished = self._impl_adjust_pacing(polished, effective_reader, record_usage=False)
    except Exception as e:
        issues.append({"type": "pacing_llm_fail", "severity": "P3", "issue": str(e)})

    # 步骤 3: 规则去 AI 痕迹 (无 LLM)
    polished = self.remove_ai_gloss(polished)

    result = {
        "content": polished,
        "reader": effective_reader,
        "issues": issues,
        "chapter_num": chapter_num,
    }
    if record_usage:
        return result, usage_total
    return result

def _impl_optimize_dialogue(self, content, reader_id, record_usage) -> Any:

    prompt = build_dialogue_prompt(content, effective_reader)
