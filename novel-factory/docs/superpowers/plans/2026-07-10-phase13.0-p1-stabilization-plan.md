```typescript
        // never resolve within timeout

    await expect(promise).rejects.toThrow(/超时/);
    expect(fetchSpy).toHaveBeenCalled();
  });

  it('caller-provided AbortSignal cancels request', async () => {
    const controller = new AbortController();
    const fetchSpy = vi.spyOn(global, 'fetch').mockResolvedValue(
      new Response('{}', { status: 200 }) as any
    );

    const promise = request('/test', { signal: controller.signal });

      expect.any(String),
      expect.objectContaining({ signal: expect.any(AbortSignal) }),
```

```python
def get_ripple_impact_scores_bulk(self, ripple_ids: list[str]) -> dict[str, float]:
    """Phase 13.0: 一次预取多 ripple 的 impact score."""
    if not ripple_ids:
        return {}
    placeholders = ",".join("?" * len(ripple_ids))
    with self._connect() as conn:
        rows = conn.execute(
            f"SELECT ripple_id, impact_score FROM ripple_impact_scores"
            f" WHERE ripple_id IN ({placeholders})",
            ripple_ids,
        ).fetchall()
    return {row["ripple_id"]: row["impact_score"] for row in rows}
```
