> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 把 budget 从内存 in-memory (Phase 8.8) 升级为 SQLite 持久化,新增 per-day / per-week 两档,AgentComputeFn 一次性检查 3 档超阈即 raise,Sidebar 优先显示最紧迫的 budget。

### 4.2 Schema (single `budgets` table, append-only)

```sql
CREATE TABLE IF NOT EXISTS budgets (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    scope TEXT NOT NULL CHECK(scope IN ('run', 'day', 'week')),
    usd REAL NOT NULL CHECK(usd >= 0),
    run_id TEXT,                    -- NULL for day/week (跨 run 共享), UUID hex for run
    set_at TEXT NOT NULL            -- UTC ISO8601
);

CREATE INDEX IF NOT EXISTS idx_budgets_scope_set_at ON budgets(scope, set_at DESC);
CREATE INDEX IF NOT EXISTS idx_budgets_run_id ON budgets(run_id);  -- NULL allowed
```

**Append-only 语义**: 每次 set 插新行 (含 run_id),current = `ORDER BY id DESC LIMIT 1 WHERE scope=?`。"无" = 该 scope 0 行。"修改" = 插新行,旧行变历史。

### 4.3 Data flow (3-tier enforce)

```
[Run start]
  master.run_workflow(cost_budget_usd=X)
  ├─ run_id = uuid.uuid4().hex
  ├─ budget_service.set(scope='run', usd=X, run_id=run_id)  ← per-run 持久化
  └─ self._current_budget_usd = X                            ← in-memory 同步 (Phase 8.8)

[Each AgentComputeFn call]
  compute(node, inputs)
  ├─ run LLM, get output/usage
  ├─ total_cost = self._cost_tracker.total_cost  (in-memory)
  └─ IF budget_service:
      └─ budget_service.check_all_scopes(total_cost)  ← check 3 档
          ├─ 'run' 档: WHERE scope='run' AND usd IS NOT NULL → used_pct = total_cost / usd
          ├─ 'day' 档: WHERE scope='day' ORDER BY id DESC LIMIT 1
          │           + date(set_at) = date('now') → 仍 active
          └─ 'week' 档: WHERE scope='week' ORDER BY id DESC LIMIT 1
                        + date(set_at, 'weekday 0', '-7 days') = date('now') → 仍 active
      └─ 任一档 > 100% raise CostBudgetExceeded(scope, used_usd, budget_usd)

[API setter]
  PUT /api/budgets/{day,week} body={usd: 0.5}
  → _handle_set_budget(scope, usd) → budget_service.set(scope, usd, run_id=None)
  → 200 OK + GET /api/budgets 返 current 3 档

[Dashboard Sidebar]
  GET /api/workflows/active/status
  → protocols.py:_extract_budget_per_window(controller, scope)
  → master.cost_budget_status (per-run, Phase 8.8) + 2 new fields (day/week)
  → SidebarCostBanner.vue allBudgets = [per-run, per-day, per-week]
  → sort by used_pct desc, exceeded 排最前
  → render activeBudget
```

### 4.4 Calendar window SQL (UTC)

| Scope | "current" 查询 | 含义 |
|-------|----------------|------|
| `run`  | `WHERE scope='run' ORDER BY id DESC LIMIT 1` | 最近一次 set |
| `day`  | `WHERE scope='day' ORDER BY id DESC LIMIT 1` + `date(set_at) = date('now')` | UTC 00:00-23:59 |
| `week` | `WHERE scope='week' ORDER BY id DESC LIMIT 1` + `date(set_at, 'weekday 0', '-7 days') = date('now')` | Mon-Sun (SQLite `weekday 0` = Sunday, `-7 days` 推到本周一) |

**Window 计算 Python-side** (避免 SQL 复杂):
```python
def _is_within_window(set_at_iso: str, scope: str) -> bool:
    set_at = datetime.fromisoformat(set_at_iso)
    now = datetime.utcnow()
    if scope == "day":
        return set_at.date() == now.date()
    elif scope == "week":
        days_since_monday = now.weekday()
        week_start = now - timedelta(days=days_since_monday)
        return set_at >= week_start.replace(hour=0, minute=0, second=0, microsecond=0)
    return True  # 'run' 不需 window
```

### 4.6 SidebarCostBanner priority cascade

```javascript
const allBudgets = computed(() => {
  const list = [];
  if (props.status?.cost_budget_status?.budget_usd) {
    list.push({ ...props.status.cost_budget_status, scope: 'run', label: '本次' });
  }
  if (props.status?.budget_per_day?.budget_usd) {
    list.push({ ...props.status.budget_per_day, scope: 'day', label: '今日' });
  }
  if (props.status?.budget_per_week?.budget_usd) {
    list.push({ ...props.status.budget_per_week, scope: 'week', label: '本周' });
  }
  return list;
});

const activeBudget = computed(() => {
  const filtered = allBudgets.value;
  if (filtered.length === 0) return null;
  // Sort: exceeded 优先, 同状态按 used_pct desc
  return [...filtered].sort((a, b) => {
    if (a.status === 'exceeded' && b.status !== 'exceeded') return -1;
    if (b.status === 'exceeded' && a.status !== 'exceeded') return 1;
    return b.used_pct - a.used_pct;
  })[0];
});
```

**Empty case**: `activeBudget.value === null` → 整组 banner 隐藏 (跟 Phase 8.11 一致)。

### 4.7 API endpoints

```python
# GET /api/budgets
@app.get("/api/budgets")
async def get_budgets() -> dict:
    """返 3 档 current budget + status"""
    return {
        "per_run": _extract_budget_status(controller),  # Phase 8.8 helper
        "per_day": _extract_budget_per_window(controller, "day"),
        "per_week": _extract_budget_per_window(controller, "week"),
    }

# PUT /api/budgets/day  body {"usd": 0.5}
# PUT /api/budgets/week body {"usd": 2.0}
class BudgetSetRequest(BaseModel):
    usd: float = Field(ge=0, le=10000)  # 0 = 无限 (但仍写行 0), > 0 = active

@app.put("/api/budgets/{scope}")
async def set_budget(scope: str, req: BudgetSetRequest) -> dict:
    if scope not in ("day", "week"):
        raise HTTPException(400, "scope must be 'day' or 'week'")
    budget_service = controller.budget_service
    if budget_service is None:
        raise HTTPException(503, "budget service not initialized")
    budget_service.set(scope, req.usd, run_id=None)
    return {"ok": True, "scope": scope, "usd": req.usd}
```

**No per-run setter API** (per-run 是 ephemeral,run 启动时设,无需 API)。**No DELETE** (YAGNI, usd=0 仍写行 0 表示 active,但 dashboard UI 暂不暴露 0;显式 "无" = 该 scope 0 行,跟 Phase 8.8 一致)。
