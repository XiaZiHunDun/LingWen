### D1 — H10 前端 API timeout/Abort

```javascript
// dashboard/frontend/src/api/index.js: 重构 request()

const DEFAULT_TIMEOUT_MS = 15_000;

export async function request(path, options = {}) {
  const { signal: externalSignal, timeout = DEFAULT_TIMEOUT_MS, ...rest } = options;
  
  // 双 signal: 外层 caller 提供 + 内层 timeout
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(new Error('timeout')), timeout);
  
  // 若 caller 传了 signal, 关联 abort
  if (externalSignal) {
    if (externalSignal.aborted) controller.abort(externalSignal.reason);
    else externalSignal.addEventListener('abort', () => controller.abort(externalSignal.reason), { once: true });
  }
  
  try {
    const response = await fetch(`/api${path}`, {
      ...rest,
      signal: controller.signal,
    });
    clearTimeout(timeoutId);
    
    // Phase 13.0: 优先解析结构化 JSON error（FastAPI `detail` 字段）
    if (!response.ok) {
      let body = null;
      try {
        body = await response.json();
      } catch {
        body = await response.text().catch(() => '');
      }
      const err = new Error(body?.detail || body || response.statusText);
      err.status = response.status;
      err.body = body;
      throw err;
    }
    
    return await response.json();
  } catch (e) {
    clearTimeout(timeoutId);
    // Phase 13.0: AbortError 转友好文案
    if (e.name === 'AbortError' || controller.signal.aborted) {
      const friendly = new Error(`请求超时 (${timeout / 1000}s) — 后端可能未运行或响应过慢`);
      friendly.code = 'TIMEOUT';
      throw friendly;
    }
    throw e;
  }
}
```

**所有 `fetch*` export 透传 pattern**:
```javascript

// caller 传: fetchXxx(id, { signal: myController.signal }) 即可
```

### D3 — H4 Ripple list N+1

```python
# dashboard/app.py: _ripple_list_items 重构
def _ripple_list_items(storage, ripples):
    """Phase 13.0: 一次预取所有 ripple 的 impact score, 消除 N+1。
    
    Before: 200 ripple → 400 DB query (每 ripple 调一次 _ripple_impact_score)
    After:  200 ripple → 2 DB query (1 次 ripples + 1 次 scores)
    """
    if not ripples:
        return []
    
    ripple_ids = [r.id for r in ripples]
    
    # Phase 13.0: 1 次 WHERE ripple_id IN (...) 预取
    impact_scores = storage.get_ripple_impact_scores_bulk(ripple_ids)
    
    return [
        _ripple_list_item_from(r, impact_scores.get(r.id, 0.0))
        for r in ripples
    ]


# infra/cross_volume/storage.py: 新增 bulk method
def get_ripple_impact_scores_bulk(self, ripple_ids: list[str]) -> dict[str, float]:
    """Phase 13.0: 一次预取多 ripple 的 impact score.
    
    Returns:
        {ripple_id: impact_score} (缺失的 ripple 不在 dict 里, caller 默认 0.0)
    """
    if not ripple_ids:
        return {}
    placeholders = ",".join("?" * len(ripple_ids))
    with self._connect() as conn:
        rows = conn.execute(
            f"SELECT ripple_id, impact_score FROM ripple_impact_scores"
            f" WHERE ripple_id IN ({placeholders})",
            ripple_ids,
        ).fetchall()
    return {row["ripple_id"]: row["impact_score"] for row in rows}
```

**关键改动**：
- `_ripple_list_items` 内部从循环 `storage._ripple_impact_score(r)` 改成一次 bulk 预取 + dict lookup
- `get_ripple_impact_scores_bulk` 是 storage 层新 method，独立可测
- 既有的 `_ripple_impact_score(r)` 单查 method 保留（其他地方可能用到）

# infra/cli/path_utils.py (NEW): 路径解析 helper
"""Phase 13.0: 统一 CLI 路径解析, 优先 $LINGWEN_PROJECT_ROOT, CWD fallback + WARNING."""
import os
import sys
from pathlib import Path


def resolve_project_db_path(
    db_filename: str,  # 例: "ripple.db"
    *,
    env_var: str = "LINGWEN_PROJECT_ROOT",
    cwd_fallback: bool = True,
) -> Path:
    """Phase 13.0: 解析项目 db 路径.
    
    Priority:
    1. $LINGWEN_PROJECT_ROOT/.state/{db_filename} (env 设了)
    2. CWD-relative ./{db_filename} (env 未设, cwd_fallback=True, WARNING)
    
    Args:
        db_filename: db 文件名 (例 "ripple.db", "cost_tracker.db")
        env_var: env var 名 (默认 LINGWEN_PROJECT_ROOT)
        cwd_fallback: 是否允许 CWD fallback (默认 True, Phase 13.0 过渡)
    
    Returns:
        解析后的 Path (不存在也可能, caller 自己判断)
    
    Raises:
        SystemExit: env 设了但 db 不存在 → exit 2 + 友好提示
    """
    root = os.environ.get(env_var)
    if root:
        candidate = Path(root) / ".state" / db_filename
        if not candidate.exists():
            print(
                f"[ERROR] {env_var}={root} 但 {candidate} 不存在. "
                f"请检查项目根路径是否正确, 或 unset {env_var} 走 CWD 模式.",
                file=sys.stderr,
            )
            sys.exit(2)
        return candidate
    if cwd_fallback:
        print(
            f"[WARNING] {env_var} 未设, 走 CWD-relative 路径 ./{db_filename}. "
            f"Phase 14 起将 exit 2, 请尽快设置 {env_var}.",
            file=sys.stderr,
        )
        return Path(f".state/{db_filename}")
    print(f"[ERROR] {env_var} 未设, 不允许 CWD fallback.", file=sys.stderr)
    sys.exit(2)
