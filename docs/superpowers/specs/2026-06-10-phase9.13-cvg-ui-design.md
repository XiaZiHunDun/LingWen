```python
class CvgConnectionManager:
    """WebSocket 连接管理器 (CVG 专用)."""
    def __init__(self) -> None:
        self._connections: set[WebSocket] = set()
        self._lock = asyncio.Lock()

    async def connect(self, ws: WebSocket) -> None: ...
    async def disconnect(self, ws: WebSocket) -> None: ...
    async def broadcast(self, event: dict[str, Any]) -> None: ...
    async def send_to(self, ws: WebSocket, event: dict[str, Any]) -> None: ...


# dashboard/app.py (additive, +30L WS endpoint)

@app.websocket("/api/ws/cvg")
async def ws_cvg(websocket: WebSocket) -> None:
    """实时推送 ripple 变化 (跟 /api/ws/workflows 1:1)."""
    await cvg_manager.connect(websocket)
    try:
        while True:
            msg = await websocket.receive_json()
            if msg.get("type") == "ping":
                await websocket.send_json({"type": "pong"})
    except WebSocketDisconnect:
        await cvg_manager.disconnect(websocket)
```

**Message types**:
- Server → Client: `{"type": "ripple_created", "data": RippleListItemResponse}` / `{"type": "ripple_status_changed", "data": {ripple_id, new_status, actor, applied_at}}` / `{"type": "pong"}`
- Client → Server: `{"type": "ping"}`

**Broadcast trigger** (additive hook in RippleStorage):
- `update_ripple_status()` 完成后 → `cvg_ws.broadcast({"type": "ripple_status_changed", "data": ...})`
- `append_nodes_atomic()` 完成后 (Phase 9.12 已就位, 0 改 storage API, 内部加 hook) → `cvg_ws.broadcast({"type": "ripple_created", "data": ...})` for each new ripple

#### 4.3.3 `useRippleStore` composable (跟 useDecisionStore 1:1 mirror)

```javascript
// dashboard/frontend/src/composables/useRippleStore.js (NEW, ~70L)

let _store = null  // module-level singleton

function createRippleStore() {
  const ripples = ref([])
  const loading = ref(false)
  const lastError = ref(null)
  const stats = ref({ total: 0, by_status: {}, by_dimension: {} })

  async function refresh({ status, dimension, volume } = {}) {
    loading.value = true
    lastError.value = null
    try {
      const params = new URLSearchParams()
      if (status) params.set('status', status)
      if (dimension) params.set('dimension', dimension)
      if (volume) params.set('volume', String(volume))
      const data = await fetchRipples(params)  // api/index.js 新增
      ripples.value = data.ripples
      stats.value = data.stats
    } catch (e) {
      lastError.value = e.message
    } finally {
      loading.value = false
    }
  }

  async function apply(rippleId) {
    try {
      const updated = await applyRipple(rippleId)  // api/index.js 新增
      const idx = ripples.value.findIndex(r => r.ripple_id === rippleId)
      if (idx >= 0) ripples.value[idx].status = updated.status
      return updated
    } catch (e) {
      lastError.value = e.message
      throw e
    }
  }

  async function reject(rippleId, reason) { /* 跟 apply 镜像 */ }

  function applySocketUpdate(update) {
    if (update.type === 'ripple_created') {
      ripples.value.unshift(update.data)
    } else if (update.type === 'ripple_status_changed') {
      const idx = ripples.value.findIndex(r => r.ripple_id === update.data.ripple_id)
      if (idx >= 0) ripples.value[idx].status = update.data.new_status
    }
  }

  return { ripples, loading, lastError, stats, refresh, apply, reject, applySocketUpdate }
}

export function useRippleStore() {
  if (!_store) _store = createRippleStore()
  return _store
}
```

#### 4.3.4 `useRippleSocket` composable (跟 useWorkflowSocket 1:1 mirror)

```javascript
// dashboard/frontend/src/composables/useRippleSocket.js (NEW, ~60L)

const RECONNECT_BASE_MS = 1000
const RECONNECT_FACTOR = 2
const RECONNECT_MAX_MS = 8000
const MAX_RECONNECT_ATTEMPTS = 3

export function useRippleSocket() {
  const ws = ref(null)
  const connected = ref(false)
  const lastError = ref(null)
  const pendingUpdates = ref([])
  const reconnectAttempts = ref(0)

  function connect() {
    ws.value = new WebSocket(`ws://${location.host}/api/ws/cvg`)
    ws.value.onopen = () => { connected.value = true; reconnectAttempts.value = 0 }
    ws.value.onmessage = (event) => {
      const update = JSON.parse(event.data)
      if (update.type === 'ping') { ws.value.send(JSON.stringify({type: 'pong'})); return }
      pendingUpdates.value.push(update)
    }
    ws.value.onclose = () => {
      connected.value = false
      if (reconnectAttempts.value < MAX_RECONNECT_ATTEMPTS) {
        const delay = Math.min(RECONNECT_BASE_MS * (RECONNECT_FACTOR ** reconnectAttempts.value), RECONNECT_MAX_MS)
        reconnectAttempts.value++
        setTimeout(connect, delay)
      }
    }
    ws.value.onerror = (e) => { lastError.value = e.message }
  }

  function disconnect() { ws.value?.close() }

  return { ws, connected, lastError, pendingUpdates, connect, disconnect, reconnectAttempts }
}
```

### 4.5 Error Handling (失败场景 + 降级 + 边界)

| 场景 | UX | 触发点 | HTTP code |
|------|-----|--------|-----------|
| Backend GET 5xx | store.lastError 显示 + retry button (跟 useDecisionStore.lastError 1:1) | fetchRipples throw | 500 |
| Backend POST apply 4xx | drawer 顶部红色 banner + 不 close drawer (用户看到错误可重试) | applyRipple throw 4xx | 422 |
| ripple 已 terminal status | apply/reject 按钮 disable (pre-check) + 4xx 后错误 banner | drawer mounted 时 check status | 409 |
| Backend POST apply 5xx | drawer banner + rollback optimistic update | applyRipple throw 5xx | 500 |
| ripple_id 不存在 | drawer 显示 "Ripple not found" + close button | GET /api/cvg/ripples/{id} 404 | 404 |
| Status filter 不合法 | Pydantic 自动 regex fail (GET 422) | query param 错 | 422 |
| WS 断连 | connected.value = false + 顶部 red dot (跟 useWorkflowSocket.connected 1:1) | onclose | n/a |
| WS 断连 + 重连失败 (3次) | top banner "Real-time updates disabled, polling fallback..." + 触发 refresh() (跟 Phase 6.4 fallback 1:1) | backoff 失败 | n/a |
| WS message parse 错 | console.warn + 不 throw (跟 useWorkflowSocket 1:1 defensive) | JSON.parse fail | n/a |
| Network offline | 全 page grey-out + retry button (跟 Phase 8.11 banner 1:1) | window 'offline' event | n/a |
| Initial load 空 (0 ripple match filter) | empty state card "No ripples match current filter" + suggestion 改 filter | refresh() return [] | n/a |
| Drawer 打开但 ripple 已被其他 tab apply | drawer open 时 GET /api/cvg/ripples/{id} 重新拉 → status 变 terminal → button disable | drawer.open 时 | n/a |

**Optimistic vs server-authoritative**:
- Apply/reject 走 **optimistic + refresh()** (跟 useDecisionStore.resolve 1:1) — 立即 UI 更新, 然后 server refresh 保权威
- WS push 走 **store.applySocketUpdate** (跟 useDecisionStore.applySocketUpdate 1:1) — 直接 update 内存
