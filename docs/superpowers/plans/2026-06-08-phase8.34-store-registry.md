```js
/**
 * useDecisionStore.js — Module-level singleton store for decision history (Phase 8.34)
 *
 * Mirror useWorkflowSocket.js (Phase 6.4+8.11) module-level singleton pattern.
 * 拉 /api/decisions/all 拿全量决策历史 (resolved/deferred/cancelled), 跟
 * useWorkflowSocket 推的 pending_decisions 共存但不冲突 — 本 store 只管 all,
 * 4 mutation (refresh/resolve/defer/cancel) 走 store action.
 *
 * 用法:
 *   const { all, loading, lastError, refresh, resolve, defer, cancel } = useDecisionStore();
 *   await resolve(decisionId, 'option_a');  // → backend → 自动 refresh
 *
 * 自动:
 * - 首个 mount → 立即 refresh + 启动 lifecycle
 * - 最后一个 unmount → mountedCount 归 0 (无 active subscriber, 保留 state)
 * - mutation 失败 → lastError 设 + throw (caller 可选择性 catch)
 *
 * 状态:
 * - all: ref<Decision[]> 默认 [] (REST 拉历史, 0 接 WS pending)
 * - loading: ref<boolean> 默认 false
 * - lastError: ref<string|null> 默认 null
 *
 * 边界 (per 主公 Phase 8.34 决策):
 * - pending_decisions 仍由 useWorkflowSocket 推 (WS 权威源, Phase 6.4+)
 * - 0 改 backend, 0 改 WS protocol, 0 改 api/index.js
 */

import { onBeforeUnmount, onMounted, ref } from 'vue';
import {
  fetchAllDecisions,
  resolveDecision,
  deferDecision,
  cancelDecision,
} from '../api/index.js';

// ---- Module-level singleton state ----
const all = ref([]);
const loading = ref(false);
const lastError = ref(null);
let mountedCount = 0;

async function refresh() {
  loading.value = true;
  lastError.value = null;
  try {
    all.value = await fetchAllDecisions();
  } catch (e) {
    lastError.value = e?.message || String(e);
  } finally {
    loading.value = false;
  }
}

async function resolve(decisionId, option) {
  try {
    await resolveDecision(decisionId, option);
    await refresh();
  } catch (e) {
    lastError.value = `解决失败: ${e?.message || e}`;
    throw e;
  }
}

async function defer(decisionId, reason) {
  try {
    await deferDecision(decisionId, reason);
    await refresh();
  } catch (e) {
    lastError.value = `推迟失败: ${e?.message || e}`;
    throw e;
  }
}

async function cancel(decisionId, reason) {
  try {
    await cancelDecision(decisionId, reason);
    await refresh();
  } catch (e) {
    lastError.value = `取消失败: ${e?.message || e}`;
    throw e;
  }
}

export function useDecisionStore() {
  onMounted(() => {
    if (mountedCount === 0) refresh();
    mountedCount += 1;
  });

  onBeforeUnmount(() => {
    mountedCount = Math.max(0, mountedCount - 1);
  });

  return { all, loading, lastError, refresh, resolve, defer, cancel };
}
```

```js
/**
 * useOverviewStore.js — Module-level singleton store for overview dashboard (Phase 8.34)
 *
 * Mirror useWorkflowSocket.js (Phase 6.4+8.11) module-level singleton pattern.
 * 拉 /api/overview + /api/chapters?range=1-30 拿追读力总览数据, 跟 page-local
 * computed (chartData / statCards) 解耦, page nav 跨 mount 共享 data 缓存.
 *
 * 用法:
 *   const { overview, chapters, loading, lastError, refresh } = useOverviewStore();
 *   const chartData = computed(() => chapters.value.map(ch => ({...})))
 *
 * 自动:
 * - 首个 mount → 立即 refresh
 * - 最后一个 unmount → mountedCount 归 0
 *
 * 状态:
 * - overview: ref<OverviewResponse|null> 默认 null
 * - chapters: ref<Chapter[]> 默认 []
 * - loading: ref<boolean> 默认 false
 * - lastError: ref<string|null> 默认 null
 *
 * 边界 (per 主公 Phase 8.34 决策):
 * - 0 改 backend
 * - chartData / statCards (computed) 留 OverviewPage, 从 store ref 派生
 */

import { onBeforeUnmount, onMounted, ref } from 'vue';
import { fetchOverview, fetchChapters } from '../api/index.js';

// ---- Module-level singleton state ----
const overview = ref(null);
const chapters = ref([]);
const loading = ref(false);
const lastError = ref(null);
let mountedCount = 0;

async function refresh() {
  loading.value = true;
  lastError.value = null;
  try {
    const [o, c] = await Promise.all([
      fetchOverview(),
      fetchChapters('1-30'),
    ]);
    overview.value = o;
    chapters.value = c.chapters || [];
  } catch (e) {
    lastError.value = e?.message || String(e);
  } finally {
    loading.value = false;
  }
}
```
