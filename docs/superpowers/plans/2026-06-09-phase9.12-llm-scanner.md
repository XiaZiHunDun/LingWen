```python
# infra/cross_volume/llm_cache.py
import hashlib
import json
import logging
import os
import tempfile
from pathlib import Path


class LLMCache:
    """Phase 9.12: 内存 + JSON 落盘 cache (Q5).

    - key = SHA256(chapter_content + prompt_id + model_id)
    - 内存 dict (L1) + JSON file (L2 持久化)
    - 命中返 dict, miss 返 None
    - JSON 损坏自动删 + 重建
    """

    DEFAULT_PATH = Path.home() / ".cache" / "lingwen" / "llm_cache.json"

    def __init__(self, cache_path: Path | None = None) -> None:
        self.logger = logging.getLogger(self.__class__.__name__)
        self._mem: dict[str, dict] = {}
        self._path = cache_path or self.DEFAULT_PATH
        self._path.parent.mkdir(parents=True, exist_ok=True)
        self._load()

    @staticmethod
    def make_key(chapter_content: str, prompt_id: str, model_id: str) -> str:
        h = hashlib.sha256()
        h.update(chapter_content.encode("utf-8"))
        h.update(b"\x00")
        h.update(prompt_id.encode("utf-8"))
        h.update(b"\x00")
        h.update(model_id.encode("utf-8"))
        return h.hexdigest()

    def get(self, key: str) -> dict | None:
        return self._mem.get(key)

    def put(self, key: str, value: dict) -> None:
        self._mem[key] = value
        self._save()

    def _load(self) -> None:
        if not self._path.exists():
            return
        try:
            self._mem = json.loads(self._path.read_text(encoding="utf-8"))
        except (json.JSONDecodeError, OSError) as e:
            self.logger.warning("Cache %s corrupted, rebuilding: %s", self._path, e)
            self._path.unlink(missing_ok=True)
            self._mem = {}

    def _save(self) -> None:
        # 原子写: tempfile + rename (防半写损坏)
        fd, tmp = tempfile.mkstemp(dir=self._path.parent, prefix=".cache_", suffix=".tmp")
        try:
            with os.fdopen(fd, "w", encoding="utf-8") as f:
                json.dump(self._mem, f, ensure_ascii=False)
            os.replace(tmp, self._path)
        except Exception:
            Path(tmp).unlink(missing_ok=True)
            raise
```

# infra/cross_volume/llm_scanner.py
import json
import logging
import time
from pathlib import Path
from typing import Any
from infra.cross_volume.llm_cache import LLMCache
from infra.cross_volume.reference_graph import ReferenceNode
from infra.ai_service.cost_tracker import CostTracker
from infra.ai_service.model_tiers import ModelTier

LLM_MAX_RETRIES = 2
LLM_RETRY_BACKOFF_BASE = 1.0
LLM_RETRY_BACKOFF_FACTOR = 2.0

PROMPT_DIR = Path(__file__).parent / "llm_prompts"

DIMENSIONS = ["character", "foreshadow", "setting", "plot_point"]
PROMPT_FILES = {
    "character": "character.txt",
    "foreshadow": "foreshadow.txt",
    "setting": "setting.txt",
    "plot_point": "plot_point.txt",
}


class LLMRetryExhausted(Exception):
    """Phase 9.12: LLM 重试耗尽, 触发 fallback."""


class LLMScanner:
    """Phase 9.12: 4 维 LLM scanner."""

    def __init__(
        self,
        router: Any,
        cache: LLMCache,
        fallback_backfiller: Any,
        cost_tracker: CostTracker,
        model_tier: ModelTier = ModelTier.SONNET,
    ) -> None:
        self.logger = logging.getLogger(self.__class__.__name__)
        self.router = router
        self.cache = cache
        self.fallback = fallback_backfiller
        self.cost_tracker = cost_tracker
        self.model_tier = model_tier
        self.model_id = f"claude-{model_tier.value}"  # 简化, Phase 9.12 用
        self._prompts = {d: (PROMPT_DIR / f).read_text(encoding="utf-8")
                         for d, f in PROMPT_FILES.items()}

    def scan_chapter(
        self,
        chapter_id: int,
        chapter_content: str,
        context: dict,
    ) -> list[ReferenceNode]:
        if not chapter_content or not chapter_content.strip():
            self.logger.warning("Empty content for ch%d, skip", chapter_id)
            return []
        if len(chapter_content) > 50_000:
            self.logger.warning("ch%d content too long, truncate to 50K", chapter_id)
            chapter_content = chapter_content[:50_000]

        nodes: list[ReferenceNode] = []
        for dim in DIMENSIONS:
            try:
                dim_nodes = self._scan_one_dim(chapter_id, chapter_content, context, dim)
                nodes.extend(dim_nodes)
            except LLMRetryExhausted:
                self.logger.warning(
                    "LLM scan %s ch%d failed, fallback to rule", dim, chapter_id
                )
                fallback_nodes = self.fallback.extractors[dim].extract(
                    chapter_id, chapter_content, context
                )
                nodes.extend(fallback_nodes)
        return nodes

    def _scan_one_dim(
        self, chapter_id: int, content: str, context: dict, dim: str,
    ) -> list[ReferenceNode]:
        cache_key = LLMCache.make_key(content, PROMPT_FILES[dim], self.model_id)
        cached = self.cache.get(cache_key)
        if cached:
            return self._parse_nodes(cached["parsed"], dim, chapter_id)

        prompt = self._prompts[dim].format(chapter_content=content, context=context)
        text = self._call_with_retry(prompt)
        try:
            parsed = json.loads(text)
        except json.JSONDecodeError as e:
            self.logger.warning("LLM %s output not JSON: %s", dim, e)
            return []

        self.cache.put(cache_key, {
            "input_tokens": 0, "output_tokens": 0,  # Phase 9.12 简化, 实际由 _call_with_retry 喂 cost_tracker
            "parsed": parsed,
        })
        return self._parse_nodes(parsed, dim, chapter_id)

    def _call_with_retry(self, prompt: str) -> str:
        last_err: Exception | None = None
        for attempt in range(LLM_MAX_RETRIES + 1):
            try:
                text, usage = self.router.generate_with_usage(
                    prompt=prompt, model_tier=self.model_tier,
                    temperature=0.3, max_tokens=2000,
                )
                self.cost_tracker.record(
                    scenario="cvg_llm_scan", tier=self.model_tier,
                    input_tokens=usage["input_tokens"],
                    output_tokens=usage["output_tokens"],
                )
                return text
            except Exception as e:
                err_str = str(e)
                last_err = e
                if "401" in err_str or "403" in err_str or "429" in err_str:
                    raise LLMRetryExhausted(f"4xx: {e}") from e
                if attempt < LLM_MAX_RETRIES:
                    wait = LLM_RETRY_BACKOFF_BASE * (LLM_RETRY_BACKOFF_FACTOR ** attempt)
                    time.sleep(wait)
        raise LLMRetryExhausted(f"max retries: {last_err}")

    def _parse_nodes(self, parsed: dict, dim: str, chapter_id: int) -> list[ReferenceNode]:
        """Phase 9.12: LLM JSON → ReferenceNode 映射.
        字段映射: LLM node_id → 忽略 (id auto uuid), name/keyword/event → title,
        first_appearance_chapter/chapter/planted_chapter → chapter,
        description/summary → description, aliases/traits/category → payload dict.
        """
        nodes = []
        for n in parsed.get("nodes", []):
            try:
                chapter_n = n.get("first_appearance_chapter",
                                   n.get("chapter",
                                         n.get("planted_chapter", chapter_id)))
                nodes.append(ReferenceNode(
                    dimension=dim,
                    volume=1,
                    chapter=chapter_n,
                    title=n.get("name", n.get("keyword", n.get("event", ""))),
                    description=n.get("description", n.get("summary", "")),
                    payload={"aliases": n.get("aliases", []),
                             "traits": n.get("traits", []),
                             "category": n.get("category", "")},
                    created_by="llm_scanner",
                    confidence=n.get("confidence", 1),
                ))
            except (KeyError, TypeError, ValueError) as e:
                self.logger.warning("Skip invalid node: %s", e)
        return nodes
```

# infra/cross_volume/edge_inferrer.py
import hashlib
import json
import logging
import time
from pathlib import Path
from typing import Any
from infra.cross_volume.llm_cache import LLMCache
from infra.cross_volume.reference_graph import ReferenceNode, ReferenceEdge
from infra.ai_service.cost_tracker import CostTracker
from infra.ai_service.model_tiers import ModelTier

LLM_MAX_RETRIES = 2
LLM_RETRY_BACKOFF_BASE = 1.0
LLM_RETRY_BACKOFF_FACTOR = 2.0

PROMPT_DIR = Path(__file__).parent / "llm_prompts"
PROMPT_FILE = "edge_inference.txt"

VALID_REL_TYPES = {"mentions", "evolves", "foreshadows", "pays_off",
                    "appears_with", "causes", "conflicts_with", "supports"}


class EdgeInferrer:
    """Phase 9.12: 8 边类型 LLM 推断."""

    def __init__(
        self,
        router: Any,
        cache: LLMCache,
        cost_tracker: CostTracker,
        model_tier: ModelTier = ModelTier.SONNET,
        confidence_threshold: int = 3,
    ) -> None:
        self.logger = logging.getLogger(self.__class__.__name__)
        self.router = router
        self.cache = cache
        self.cost_tracker = cost_tracker
        self.model_tier = model_tier
        self.model_id = f"claude-{model_tier.value}"
        self.confidence_threshold = confidence_threshold
        self._prompt = (PROMPT_DIR / PROMPT_FILE).read_text(encoding="utf-8")

    def infer_edges(
        self,
        chapter_id: int,
        chapter_content: str,
        nodes: list[ReferenceNode],
    ) -> list[ReferenceEdge]:
        if not nodes:
            return []
        # Filter by confidence
        filtered = [n for n in nodes if n.confidence >= self.confidence_threshold]
        if not filtered:
            return []

        # Cache key based on filtered nodes + content
        nodes_dump = json.dumps([{"id": n.id, "dim": n.dimension, "conf": n.confidence}
                                  for n in filtered], sort_keys=True)
        cache_input = f"{nodes_dump}|{chapter_content}"
        cache_key = LLMCache.make_key(cache_input, PROMPT_FILE, self.model_id)
        cached = self.cache.get(cache_key)
        if cached:
            return self._parse_edges(cached["parsed"], filtered, chapter_id)

        prompt = self._prompt.format(
            nodes=nodes_dump, chapter_content=chapter_content,
        )
        try:
            text = self._call_with_retry(prompt)
        except Exception as e:
            self.logger.warning("Edge inference failed ch%d: %s", chapter_id, e)
            return []

        try:
            parsed = json.loads(text)
        except json.JSONDecodeError as e:
            self.logger.warning("Edge LLM output not JSON: %s", e)
            return []

        self.cache.put(cache_key, {
            "input_tokens": 0, "output_tokens": 0,
            "parsed": parsed,
        })
        return self._parse_edges(parsed, filtered, chapter_id)

    def _call_with_retry(self, prompt: str) -> str:
        last_err: Exception | None = None
        for attempt in range(LLM_MAX_RETRIES + 1):
            try:
                text, usage = self.router.generate_with_usage(
                    prompt=prompt, model_tier=self.model_tier,
                    temperature=0.3, max_tokens=2000,
                )
                self.cost_tracker.record(
                    scenario="cvg_edge_inference", tier=self.model_tier,
                    input_tokens=usage["input_tokens"],
                    output_tokens=usage["output_tokens"],
                )
                return text
            except Exception as e:
                last_err = e
                err_str = str(e)
                if "401" in err_str or "403" in err_str or "429" in err_str:
                    raise
                if attempt < LLM_MAX_RETRIES:
                    wait = LLM_RETRY_BACKOFF_BASE * (LLM_RETRY_BACKOFF_FACTOR ** attempt)
                    time.sleep(wait)
        raise last_err  # type: ignore

    def _parse_edges(
        self, parsed: dict, nodes: list[ReferenceNode], chapter_id: int,
    ) -> list[ReferenceEdge]:
        """Phase 9.12: LLM JSON → ReferenceEdge 映射.
        字段映射: LLM from_id/to_id → dataclass from_node_id/to_node_id,
        weight → weight (float, 校验 [0.0, 1.0]), evidence → evidence.
        """
        valid_ids = {n.id for n in nodes}
        edges = []
        for i, e in enumerate(parsed.get("edges", [])):
            try:
                if e["from_id"] not in valid_ids or e["to_id"] not in valid_ids:
                    continue
                if e["from_id"] == e["to_id"]:
                    continue  # self-loop
                if e["relationship_type"] not in VALID_REL_TYPES:
                    continue
                edges.append(ReferenceEdge(
                    from_node_id=e["from_id"],
                    to_node_id=e["to_id"],
                    relationship_type=e["relationship_type"],
                    weight=float(e["weight"]),
                    created_by="edge_inferrer",
                    evidence=e.get("evidence", ""),
                ))
            except (KeyError, TypeError, ValueError) as err:
                self.logger.warning("Skip invalid edge: %s", err)
        return edges
```
