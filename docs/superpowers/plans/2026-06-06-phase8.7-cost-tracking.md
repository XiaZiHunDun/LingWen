def _impl_polish_merge_synthesis(
    self,
    content_a: str,
    content_b: str,
    *,
    labels: tuple[str, str] = ("A", "B"),
    record_usage: bool,
) -> tuple[Dict[str, Any], Dict[str, int]] | Dict[str, Any]:
    """Phase 7.5 baseline + Phase 8.7 record_usage kwarg.

    Returns:
        - record_usage=True: tuple[dict, {"input_tokens": int, "output_tokens": int}]
        - record_usage=False: dict (旧契约, backward compat)
    """
    empty_usage: Dict[str, int] = {"input_tokens": 0, "output_tokens": 0}

    if not content_a or not content_b:
        result = self._merge_synthesis_len_fallback(content_a, content_b, labels=labels, reason="empty_content")
        if record_usage:
            return result, empty_usage
        return result

    if content_a == content_b:
        result = {
            "content": content_a,
            "winner": labels[0],
            "scores_a": {},
            "scores_b": {},
            "scores_total_a": 0.0,
            "scores_total_b": 0.0,
            "scores_delta": 0.0,
            "fallback": "identical",
        }
        if record_usage:
            return result, empty_usage
        return result

    # LLM path
    prompt = build_merge_synthesis_prompt(content_a, content_b, labels=labels)
    system = get_merge_synthesis_system_prompt()
    try:
        if record_usage:
            response, usage = self.polisher.chat_with_usage(
                prompt=prompt, system=system,
                temperature=0.2, max_tokens=2000,
            )
        else:
            response = self.polisher.chat(
                prompt=prompt, system=system,
                temperature=0.2, max_tokens=2000,
            )
            usage = empty_usage
        parsed = self._parse_merge_response(response)
    except _MergeParseError:
        result = self._merge_synthesis_len_fallback(content_a, content_b, labels=labels, reason="json_parse_failed")
        if record_usage:
            return result, empty_usage
        return result
    except Exception as e:
        logger.warning(f"polish_merge_synthesis LLM 失败: {e}", exc_info=True)
        result = self._merge_synthesis_len_fallback(content_a, content_b, labels=labels, reason="llm_fail")
        if record_usage:
            return result, empty_usage
        return result

    scores_a = parsed["scores_a"]
    scores_b = parsed["scores_b"]
    total_a = sum(_coerce_score(scores_a.get(k, 5)) for k in _S1_S8_KEYS) / 8.0
    total_b = sum(_coerce_score(scores_b.get(k, 5)) for k in _S1_S8_KEYS) / 8.0
    winner = labels[0] if total_a >= total_b else labels[1]
    result = {
        "content": content_a if winner == labels[0] else content_b,
        "winner": winner,
        "scores_a": {k: _coerce_score(scores_a.get(k, 5)) for k in _S1_S8_KEYS},
        "scores_b": {k: _coerce_score(scores_b.get(k, 5)) for k in _S1_S8_KEYS},
        "scores_total_a": total_a,
        "scores_total_b": total_b,
        "scores_delta": (total_a - total_b) / 8.0,
    }
    if record_usage:
        return result, usage
    return result
