```python
def trigger_cascade(
    self, ripple: CrossVolumeRipple, max_depth: int = 3, weighted: bool = True,
) -> CascadedRipple:
    """Phase 9.15: real BFS cascade (replaces Phase 9.10 stub returns []).
    Phase 9.16: weighted BFS using heapq priority queue (high edge.weight first).

    Args:
        ripple: Trigger ripple (起点节点集)
        max_depth: BFS 深度上限 (default 3, Phase 9.15 既设)
        weighted: True = v2_weighted (heapq, 高 weight 优先, ties node_id 字典序),
                  False = v1 FIFO (deque, 跟 Phase 9.15 完全等价, backward compat)

    Returns:
        CascadedRipple with bfs_algorithm_version='v2_weighted' (weighted=True)
        or 'v1' (weighted=False).
    """
    algorithm_version = "v2_weighted" if weighted else "v1"

    if weighted:
        # Phase 9.16: heapq priority queue, tuple (-weight, node_id, node, depth)
        # -weight 让高 weight 优先 pop (heapq 默认最小堆)
        # node_id 作 tie-breaker 保稳定 (Python tuple 字典序)
        pq: list[tuple[float, str, ReferenceNode, int]] = []
        for n in ripple.affected_nodes:
            node = self._nodes.get(n)
            if node is None:
                continue
            # Phase 9.16: 起点的 weight 用 1.0 (最高, 实际仅 visited 防 revisit)
            heapq.heappush(pq, (-1.0, n, node, 1))
        visited: set[str] = set(ripple.affected_nodes)
    else:
        # Phase 9.15: FIFO deque (backward compat)
        from collections import deque
        queue: deque[tuple[ReferenceNode, int]] = deque(
            (self._nodes[n], 1) for n in ripple.affected_nodes if n in self._nodes
        )
        visited = set(ripple.affected_nodes)

    collected_nodes: list[ReferenceNode] = []
    collected_edges: list[ReferenceEdge] = []
    collected_actions: list[dict] = []
    reached_depth = 0

    while True:
        if weighted:
            if not pq:
                break
            _, _, node, depth = heapq.heappop(pq)
        else:
            if not queue:
                break
            node, depth = queue.popleft()

        if depth > max_depth:
            continue

        reached_depth = max(reached_depth, depth)

        # 收集本 hop 节点 (起点不收集)
        if node.id not in (set(ripple.affected_nodes) if not weighted else visited) or depth > 1:
            # 简化: 仅收集 depth > 1 的节点 (起点不算 cascade)
            pass

        if depth == 1:
            # 起点不收集
            pass
        else:
            collected_nodes.append(node)

        # BFS 邻居
        neighbor_ids = set()
        for eid in self._index_by_node_edges.get(node.id, set()):
            e = self._edges[eid]
            other = e.to_node_id if e.from_node_id == node.id else e.from_node_id
            neighbor_ids.add(other)

        for nid in neighbor_ids:
            if nid in visited:
                continue
            visited.add(nid)
            neighbor = self._nodes.get(nid)
            if neighbor is None:
                continue
            # 找 edge
            edge_obj = None
            for eid in self._index_by_node_edges.get(node.id, set()):
                e = self._edges[eid]
                if (e.from_node_id == node.id and e.to_node_id == nid) or (
                    e.from_node_id == nid and e.to_node_id == node.id
                ):
                    edge_obj = e
                    break
            if edge_obj is not None and depth < max_depth:
                collected_edges.append(edge_obj)
            collected_actions.append({
                "node_id": nid, "depth": depth, "action": "modify",
                "weight": getattr(edge_obj, "weight", 0.5) if edge_obj else 0.5,
            })
            if depth < max_depth:
                weight = getattr(edge_obj, "weight", 0.5) if edge_obj else 0.5
                if weighted:
                    heapq.heappush(pq, (-weight, nid, neighbor, depth + 1))
                else:
                    queue.append((neighbor, depth + 1))

    return CascadedRipple(
        trigger_ripple_id=ripple.id,
        cascade_nodes=tuple(collected_nodes),
        cascade_edges=tuple(collected_edges),
        cascade_actions=tuple(collected_actions),
        depth_reached=reached_depth,
        generated_at=datetime.utcnow().isoformat() + "Z",
        bfs_algorithm_version=algorithm_version,
    )
```

```js
    const data = JSON.parse(event.data);
    if (data.type === 'connected') {
      if (data.snapshot) status.value = data.snapshot;
      if (Array.isArray(data.pending_decisions)) {
        pendingDecisions.value = data.pending_decisions;
      }
    } else if (data.type === 'workflow.status' && data.payload) {
      status.value = data.payload;
      registeredHandlers.status.forEach((h) => h(data.payload));
    } else if (data.type === 'decision.snapshot' && Array.isArray(data.payload)) {
      pendingDecisions.value = data.payload;
      registeredHandlers.pendingDecisions.forEach((h) => h(data.payload));
    } else if (data.type === 'cascade.update' && data.payload) {  // Phase 9.16
      latestCascadeUpdates.value = [...latestCascadeUpdates.value, data.payload].slice(-10);
      registeredHandlers.cascade.forEach((h) => h(data.payload));
    }
  } catch (e) {

  }

export function onCascadeUpdate(handler) {
```
