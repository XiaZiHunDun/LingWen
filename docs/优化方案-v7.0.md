from datetime import datetime, timedelta

STALE_THRESHOLD = timedelta(minutes=30)

def check_stale_tasks(workflow_file):
    with open(workflow_file, 'r') as f:
        wf = json.load(f)

    for task_id, task in wf.get('agent_tasks', {}).items():

### P0-3：Python fallback 无锁

**问题**：jq不可用时，整个并发保护完全失效

**修复方案**：

```bash
# run_workflow.sh 第131-156行 jq_get 函数修改
function jq_get() {
    local key=$1
    local fallback=${2:-""}

    if [ "$JQ_AVAILABLE" = true ]; then
        jq -r "$key" "$WORKFLOW_FILE" 2>/dev/null || echo "$fallback"
    else
        # Python fallback - 使用fcntl.flock
        python3 << PYEOF
import fcntl
import json
import os

lock_file = '$PROJECT_ROOT/.locks/workflow.lock'
os.makedirs(os.path.dirname(lock_file), exist_ok=True)

with open(lock_file, 'w') as lf:
    fcntl.flock(lf.fileno(), fcntl.LOCK_EX)
    try:
        with open('$WORKFLOW_FILE', 'r', encoding='utf-8') as f:
            data = json.load(f)
        # 提取key
        keys = '$key'.replace('.', "']['").replace('.[', "][").replace('$", "")
        result = data
        for k in keys.split(']['):
            if k.startswith('['):
                result = result[int(k[1:-1])]
            else:
                result = result[k] if k in result else '$fallback'
        print(result if result else '$fallback')
    finally:
        fcntl.flock(lf.fileno(), fcntl.LOCK_UN)
PYEOF
    fi
}
```

**关键变更**：
- 移除 `/tmp/lingwen_workflow.lock` → 改用 `$PROJECT_ROOT/.locks/workflow.lock`
- Python fallback 时使用 `fcntl.flock` 替代无锁

### P1-1：伏笔回收率仅40.7%

**问题**：万剑宗相关23个core级伏笔未回收，是最大"吃书"集中区

**当前检测器**：`consistency/checkers/foreshadow_checker.py`

```python
# novel-factory/consistency/checkers/core_foreshadow_checker.py（新增）

class CoreForeshadowChecker:
    """
    核心伏笔检测器 - 强制100%回收
    """

    def check(self, chapters: list[Chapter], foreshadow_records: list[dict]) -> list[Issue]:
        issues = []
        core_level = [f for f in foreshadow_records if f.get('level') == 'core']

        for foreshadow in core_level:
            # 检查是否回收
            found = any(
                self._is_recycled(foreshadow, ch.content)
                for ch in chapters
            )

            if not found:
                issues.append(Issue(
                    severity='HIGH',
                    chapter=foreshadow['chapter'],
                    type='unrecycled_foreshadow',
                    description=f"Core级伏笔未回收: {foreshadow['foreshadow_text']}",
                    expect_recycled_by=foreshadow.get('expect_recycled_by', [])
                ))

        return issues

    def _is_recycled(self, foreshadow: dict, content: str) -> bool:
        # 检测伏笔是否被回收（关键词匹配+语义相似度）
        keywords = foreshadow.get('keywords', [])
        return any(kw in content for kw in keywords)
```

}

### P1-3：模板句密度过高

**问题**：104个章节被标记为"高重复"，文风机械

**当前实现**：`tools/consistency/check_template_sentences.py`

**修复方案**：增加多样性评分触发重写

```python
# novel-factory/consistency/checkers/sentence_diversity_checker.py（新增）

class SentenceDiversityChecker:
    """
    句式多样性检测器
    评分标准：
    - 优秀: 句式种类≥10种，且无单一句式超过20%
    - 合格: 句式种类≥6种
    - 触发重写: 句式种类<6种 或 某句式占比>40%
    """

    DIVERSE_PATTERNS = [
        r'「.+」',           # 对话
        r'他.+，.+。',       # 他述句
        r'突然.+',           # 副词开头
        r'然而.+',           # 转折句
        r'只见.+',           # 观察句
        r'随着.+',           # 时间句
    ]

    def score_chapter(self, content: str) -> float:
        # 统计句式种类和分布
        patterns = [re.compile(p) for p in self.DIVERSE_PATTERNS]

        distribution = {}
        for pattern in patterns:
            matches = pattern.findall(content)
            if matches:
                distribution[pattern.pattern] = len(matches)

        # 计算多样性指数
        total_sentences = len(re.findall(r'[。！？]', content))
        if total_sentences == 0:
            return 0.0

        # Shannon多样性指数
        diversity_index = 0
        for count in distribution.values():
            p = count / total_sentences
            if p > 0:
                diversity_index -= p * math.log2(p)

        return diversity_index

    def check_threshold(self, chapters: list[Chapter]) -> list[Issue]:
        issues = []
        for ch in chapters:
            score = self.score_chapter(ch.content)
            if score < 3.0:  # 阈值
                issues.append(Issue(
                    severity='MEDIUM',
                    chapter=ch.number,
                    type='low_sentence_diversity',
                    description=f"句式多样性评分{score:.2f}，低于阈值3.0"
                ))
        return issues
```

**修改现有检查器**：更新 `check_template_sentences.py` 调用新的多样性检测

### P1-4：opinion resolve 无权限校验

**问题**：任何人可标记意见为"已解决"，无主编签字确认

**涉及文件**：`06_意见仓库/run_opinion.sh:254-310`

**修复方案**：增加主编签字验证

```bash
# run_opinion.sh cmd_resolve 函数修改

cmd_resolve() {
    local opinion_id=${1:-""}

    if [ -z "$opinion_id" ]; then
        log_error "请指定意见ID"
        exit 1
    fi

    # 解析意见文件路径
    local opinion_file="$OPINION_DIR/$(echo $opinion_id | cut -d'_' -f1)/${opinion_id}.md"

    if [ ! -f "$opinion_file" ]; then
        log_error "意见文件不存在: $opinion_file"
        exit 1
    fi

    # 检查主编签字
    if ! grep -q "## 主编签字" "$opinion_file"; then
        log_error "缺少主编签字，无法解决意见"
        log_error "请在意见文件中添加: ## 主编签字"
        exit 1
    fi

    # 检查签字人
    local signer=$(grep -A1 "## 主编签字" "$opinion_file" | tail -1 | tr -d ' ')
    if [ -z "$signer" ] || [ "$signer" == "待签字" ]; then
        log_error "主编未签字"
        exit 1
    fi

    # 执行解决
    local timestamp=$(date '+%Y-%m-%d')
    echo "" >> "$opinion_file"
    echo "**解决状态**: 已解决 @ $timestamp by $signer" >> "$opinion_file"

    log_info "意见 $opinion_id 已解决"
}
```

**SQLite事务封装**：

# database.py
import sqlite3
from contextlib import contextmanager

class WorkflowDB:
    def __init__(self, db_path):
        self.db_path = db_path
        self._init_db()

    @contextmanager
    def transaction(self):
        conn = sqlite3.connect(self.db_path)
        conn.execute("BEGIN IMMEDIATE")
        try:
            yield conn
            conn.commit()
        except:
            conn.rollback()
            raise
        finally:
            conn.close()

    def atomic_update(self, key: str, value: dict):
        with self.transaction() as conn:
            conn.execute("""
                INSERT OR REPLACE INTO workflow_state (key, value, updated_at)
                VALUES (?, ?, CURRENT_TIMESTAMP)
            """, [key, json.dumps(value)])
